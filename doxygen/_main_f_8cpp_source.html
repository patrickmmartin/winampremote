62 void __fastcall TfrmMain::TrayNotify(TMessage& Msg)
63 {
64  POINT MousePos;
65 
66  switch(Msg.LParam)
67  {
68  case WM_RBUTTONDOWN:
69  // get round clicks to spurious systray windows when show
70  DoubleClickedR = false;
71  break;
72 
73  case WM_RBUTTONUP:
74  if (DoubleClickedR)
75  ShowMainForm->Execute();
76  else
77  if (GetCursorPos(&MousePos))
78  {
79  pmnuMain->PopupComponent = frmMain;
80  SetForegroundWindow(Handle);
81  pmnuMain->Popup(MousePos.x, MousePos.y);
82  }
83  break;
84 
85  case WM_RBUTTONDBLCLK:
86  DoubleClickedR = true;
87  break;
88 
89 
90  case WM_LBUTTONDOWN:
91  DoubleClickedL = false;
92  break;
93 
94  case WM_LBUTTONUP:
95  if (!DoubleClickedL)
96  Pause->Execute();
97  break;
98 
99  case WM_LBUTTONDBLCLK:
100  // as pause will have updated WAstatus for us
101  if (WAStatus == WA_NOT_PLAYING)
102  {
103  Play->Execute();
104  }
105  if (WAStatus == WA_PAUSED)
106  {
107  Stop->Execute();
108  }
109  DoubleClickedL = true;
110  break;
111 
112  default:
113  OutputDebugString(AnsiString(Msg.LParam).c_str());
114  break;
115  }
116 
117  TForm::Dispatch(&Msg);
118 
119 }

760  Repeat->Checked = (WinampVerNo >= 0x2604) && IntegerResult(IdentChars, IPC_GETREPEATOPTION, 0);

808 
809  if (frmPlaylist)

812  {
813  if (length != LastLength)
814  {
815  PlaylistRefresh->Execute();
816  }
817  }

822  {
