81 static char * StandardVersion(int version)
82 {
83 // formats the version according to (semi) standard numeric scheme
84  if (((version >> 4) & 0xf) != 0)
85  sprintf(winampver, "winamp %x.%x%x%x", version >> 12,
86  (version >> 8) & 0xf, (version >> 4) & 0xf, version & 0xf);
87  else
88  sprintf(winampver, "winamp %x.%x%x", version >> 12, (version >> 8) & 0xf,
89  version & 0xf);
90  return winampver;
91 }

93 const char * WinampVersionString(int version)
94 {
95 
96  switch (version)
97  {
98  //special cases
99  case 0x00:
100  strcpy(winampver, "Winamp unavailable");
101  break;
102  case 0x1551:
103  strcpy(winampver, "winamp 1.55");
104  break;
105  case 0x16A0:
106  strcpy(winampver, "winamp 1.6b");
107  break;
108  case 0x16AF:
109  strcpy(winampver, "winamp 1.60");
110  break;
111  case 0x16B0:
112  strcpy(winampver, "winamp 1.61");
113  break;
114  case 0x16B1:
115  strcpy(winampver, "winamp 1.62");
116  break;
117  case 0x16B3:
118  strcpy(winampver, "winamp 1.64");
119  break;
120  case 0x16B4:
121  strcpy(winampver, "winamp 1.666");
122  break;
123  case 0x16B5:
124  strcpy(winampver, "winamp 1.69");
125  break;
126 
127  default:
128  if (version > 0x2604)
129  {
130  StandardVersion(version);
131  }
132  else
133  {
134  StandardVersion(version);
135  }
136  break;
137  }
138  return winampver;
139 }
