236 void TfrmAbout::drawTorus(void)
237 {
238  struct vertex {
239  GLfloat t[2];
240  GLfloat n[3];
241  GLfloat v[3];
242  };
243 
244  int numVerts = (objectNumMajor+1) * (objectNumMinor+1);
245  int numStrips = halfObject ? objectNumMajor / 2 : objectNumMajor;
246  int numPerStrip = 2 * (objectNumMinor+1);
247  int numElements = (objectNumMajor+1) * numPerStrip;
248 
249  static struct vertex *vertexArray, *v;
250  static GLuint *elementArray, *e;
251  static int numMajor;
252  static int numMinor;
253  int i, j;
254 
255  if (!vertexArray || numMajor!=objectNumMajor || numMinor!=objectNumMinor) {
256  float majorRadius = 0.6F;
257  float minorRadius = 0.2F;
258  double majorStep = 2.0F*M_PI / objectNumMajor;
259  double minorStep = 2.0F*M_PI / objectNumMinor;
260 
261  if (vertexArray) free(vertexArray);
262  vertexArray = (struct vertex *)
263  calloc(numVerts, sizeof(struct vertex));
264 
265  if (elementArray) free(elementArray);
266  elementArray = (GLuint *)
267  calloc(numElements, sizeof(GLuint));
268 
269  numMajor = objectNumMajor;
270  numMinor = objectNumMinor;
271 
272  v = vertexArray;
273  e = elementArray;
274  for (i=0; i<=numMajor; ++i) {
275  double a = i * majorStep;
276  GLfloat x = (GLfloat) cos(a);
277  GLfloat y = (GLfloat) sin(a);
278 
279  for (j=0; j<=numMinor; ++j) {
280  double b = j * minorStep;
281  GLfloat c = (GLfloat) cos(b);
282  GLfloat r = minorRadius * c + majorRadius;
283  GLfloat z = minorRadius * (GLfloat) sin(b);
284 
285  v->t[0] = i/(GLfloat) numMajor;
286  v->t[1] = j/(GLfloat) numMinor;
287 
288  v->n[0] = x*c;
289  v->n[1] = y*c;
290  v->n[2] = z/minorRadius;
291 
292  v->v[0] = x*r;
293  v->v[1] = y*r;
294  v->v[2] = z;
295 
296  v++;
297 
298  *e++ = i * (numMinor+1) + j;
299  *e++ = (i+1) * (numMinor+1) + j;
300  }
301  }
302  }
303 
304  if (useVertexArray) {
305  glInterleavedArrays(GL_T2F_N3F_V3F, 0, vertexArray);
306 
307 #if defined(GL_SGI_compiled_vertex_array)
308  if (useVertexLocking && LockArrays) LockArrays(0, numVerts);
309 #endif
310 
311  for (i=0, e=elementArray; i<numStrips; ++i, e+=numPerStrip) {
312  glDrawElements(GL_TRIANGLE_STRIP, numPerStrip, GL_UNSIGNED_INT, e);
313  }
314 
315 #if defined(GL_SGI_compiled_vertex_array)
316  if (useVertexLocking && UnlockArrays) UnlockArrays();
317 #endif
318  } else {
319  for (i=0, e=elementArray; i<numStrips; ++i, e+=numPerStrip) {
320  glBegin(GL_TRIANGLE_STRIP);
321  for (j=0; j<numPerStrip; ++j) {
322  v = &vertexArray[e[j]];
323 
324  glTexCoord2fv(v->t);

701 void TfrmAbout::ptov(int x, int y, int width, int height, float v[3])
702 {
703  float d, a;
704 
705  // project x,y onto a hemi-sphere centered within width, height
706  v[0] = (2.0F*x - width) / width;
707  v[1] = (height - 2.0F*y) / height;
708  d = (float) sqrt(v[0]*v[0] + v[1]*v[1]);
709  v[2] = (float) cos((M_PI/2.0F) * ((d < 1.0F) ? d : 1.0F));
710  a = 1.0F / (float) sqrt(v[0]*v[0] + v[1]*v[1] + v[2]*v[2]);
711  v[0] *= a;
712  v[1] *= a;
713  v[2] *= a;
714 }

744 void TfrmAbout::trackMotion(DWORD /* time */, int x, int y)
745 {
746  if (trackingMotion) {
747  float curPos[3], dx, dy, dz;
748 
749  ptov(x, y, TheControl->ClientWidth, TheControl->ClientHeight, curPos);
750 
751  dx = curPos[0] - lastPos[0];
752  dy = curPos[1] - lastPos[1];
753  dz = curPos[2] - lastPos[2];
754  angle = 90.0F * (float) sqrt(dx*dx + dy*dy + dz*dz);
755 
756  axis[0] = lastPos[1]*curPos[2] - lastPos[2]*curPos[1];
757  axis[1] = lastPos[2]*curPos[0] - lastPos[0]*curPos[2];
758  axis[2] = lastPos[0]*curPos[1] - lastPos[1]*curPos[0];
759 
760  lastPos[0] = curPos[0];
761  lastPos[1] = curPos[1];
762  lastPos[2] = curPos[2];

768 void TfrmAbout::setupPalette(HDC hDC)
769 {
770  PIXELFORMATDESCRIPTOR pfd;
771  LOGPALETTE* pPal;
772  int pixelFormat = GetPixelFormat(hDC);
773  int paletteSize;
774 
775  DescribePixelFormat(hDC, pixelFormat, sizeof(PIXELFORMATDESCRIPTOR), &pfd);
776  if (!((pfd.dwFlags & PFD_NEED_PALETTE) ||
777  (pfd.iPixelType == PFD_TYPE_COLORINDEX)))
778  {
779  return;
780  }
781 
782  paletteSize = 1 << pfd.cColorBits;
783  pPal = (LOGPALETTE*)
784  malloc(sizeof(LOGPALETTE) + paletteSize * sizeof(PALETTEENTRY));
785  pPal->palVersion = 0x300;
786  pPal->palNumEntries = (WORD) paletteSize;
787 
788  // start with a copy of the current system palette
789  (void) GetSystemPaletteEntries(hDC, 0, paletteSize, &pPal->palPalEntry[0]);
790 
791  {
792  // fill in an RGBA color palette
793  int redMask = (1 << pfd.cRedBits) - 1;
794  int greenMask = (1 << pfd.cGreenBits) - 1;
795  int blueMask = (1 << pfd.cBlueBits) - 1;
796  int i;
797 
798  for (i=0; i<paletteSize; ++i) {
799  pPal->palPalEntry[i].peRed = (GLbyte)
800  ((((i >> pfd.cRedShift) & redMask) * 255) / redMask);
801  pPal->palPalEntry[i].peGreen = (GLbyte)
802  ((((i >> pfd.cGreenShift) & greenMask) * 255) / greenMask);
803  pPal->palPalEntry[i].peBlue = (GLbyte)
804  ((((i >> pfd.cBlueShift) & blueMask) * 255) / blueMask);
805  pPal->palPalEntry[i].peFlags = 0;
806  }
807  }
808 
809  hPalette = CreatePalette(pPal);
810  free(pPal);
811 
812  if (hPalette) {
813  SelectPalette(hDC, hPalette, FALSE);
814  RealizePalette(hDC);
815  }
816 }

818 void TfrmAbout::setupPixelformat(HDC hDC)
819 {
820  PIXELFORMATDESCRIPTOR pfd = {
821  sizeof(PIXELFORMATDESCRIPTOR), // size of this pfd
822  1, // version num
823  PFD_DRAW_TO_WINDOW | // support window
824  PFD_SUPPORT_OPENGL, // support OpenGL

826  8, // 8-bit color depth

832  0, // depth buffer (filled below)

834  0, // no auxiliary buffers
