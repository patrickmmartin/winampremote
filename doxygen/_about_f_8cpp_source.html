00701 void TfrmAbout::ptov(int x, int y, int width, int height, float v[3])
00702 {
00703     float d, a;
00704 
00705     // project x,y onto a hemi-sphere centered within width, height
00706     v[0] = (2.0F*x - width) / width;
00707     v[1] = (height - 2.0F*y) / height;
00708     d = (float) sqrt(v[0]*v[0] + v[1]*v[1]);
00709     v[2] = (float) cos((M_PI/2.0F) * ((d < 1.0F) ? d : 1.0F));
00710     a = 1.0F / (float) sqrt(v[0]*v[0] + v[1]*v[1] + v[2]*v[2]);
00711     v[0] *= a;
00712     v[1] *= a;
00713     v[2] *= a;
00714 }

00769 {
00770     PIXELFORMATDESCRIPTOR pfd;
00771     LOGPALETTE* pPal;
00772     int pixelFormat = GetPixelFormat(hDC);
00773     int paletteSize;
00774 
00775     DescribePixelFormat(hDC, pixelFormat, sizeof(PIXELFORMATDESCRIPTOR), &pfd);
00776     if (!((pfd.dwFlags & PFD_NEED_PALETTE) ||
00777           (pfd.iPixelType == PFD_TYPE_COLORINDEX)))
00778     {
00779         return;
00780     }
00781 
00782     paletteSize = 1 << pfd.cColorBits;
00783     pPal = (LOGPALETTE*)
00784         malloc(sizeof(LOGPALETTE) + paletteSize * sizeof(PALETTEENTRY));
00785     pPal->palVersion = 0x300;
00786     pPal->palNumEntries = (WORD) paletteSize;
00787 
00788     // start with a copy of the current system palette
00789     (void) GetSystemPaletteEntries(hDC, 0, paletteSize, &pPal->palPalEntry[0]);
00790 
00791     {
00792         // fill in an RGBA color palette
00793         int redMask = (1 << pfd.cRedBits) - 1;
00794         int greenMask = (1 << pfd.cGreenBits) - 1;
00795         int blueMask = (1 << pfd.cBlueBits) - 1;
00796         int i;
00797 
00798         for (i=0; i<paletteSize; ++i) {
00799             pPal->palPalEntry[i].peRed = (GLbyte)
00800                     ((((i >> pfd.cRedShift) & redMask) * 255) / redMask);
00801             pPal->palPalEntry[i].peGreen = (GLbyte)
00802                     ((((i >> pfd.cGreenShift) & greenMask) * 255) / greenMask);
00803             pPal->palPalEntry[i].peBlue = (GLbyte)
00804                     ((((i >> pfd.cBlueShift) & blueMask) * 255) / blueMask);
00805             pPal->palPalEntry[i].peFlags = 0;
00806         }
00807     }
00808 
00809     hPalette = CreatePalette(pPal);
00810     free(pPal);
