221 void __fastcall TfrmServers::GetServerIp(TObject *)
222 {
223 
224  TStringList * Addresses = new TStringList;
225  TStringList * Aliases = new TStringList;
226  AnsiString HostName;
227 
228  Screen->Cursor = crHourGlass;
229  try
230  {
231  lvMessages->Items->Clear();
232  if (lvServers->Selected)
233  {
234 
235  try
236  {
237  GetIPAddress(lvServers->Selected->Caption.c_str(), HostName, Addresses, Aliases);
238 
239  doNetworkMessage(sIPLookup + lvServers->Selected->Caption, 1);
240  doNetworkMessage(sAuthoritativeName + HostName, 1);
241 
242  if (Addresses->Count > 0 )
243  {
244  doNetworkMessage(sAddressesRetrieved, 1);
245  for (int i = 0; i < Addresses->Count ; i ++)
246  {
247  doNetworkMessage(sIPAddress + Addresses->Strings[i], 1);
248  }
249  }
250  else
251  {
252  doNetworkMessage(sAddressesNotRetrieved, 2);
253  }
254 
255 
256  if (Aliases->Count > 0 )
257  {
258  doNetworkMessage(sAliasesRetrieved, 1);
259  for (int i = 0; i < Aliases->Count ; i ++)
260  {
261  doNetworkMessage(AnsiString(sIPAlias) + Aliases->Strings[i], 1);
262  }
263  }
264  else
265  {
266  doNetworkMessage(sAliasesNotRetrieved, 2);
267  }
268 
269  }
270  catch (EIPException &E)
271  {
272  doNetworkMessage(sGetIPAddressFailed + E.Message, 3);
273  }
274 
275  }
276  }
277  __finally
278  {
279  Screen->Cursor = crDefault;
280  delete Aliases;
281  delete Addresses;
282  }
283 }
