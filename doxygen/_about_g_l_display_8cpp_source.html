89 void AboutGLDisplay::CreateFontList()
90 {
91 
92  // create text display list
93  LOGFONT lf;
94  HFONT hFont, hOldFont;
95 
96  glMatrixMode(GL_MODELVIEW);
97 
98  //sets up a display list for entire character set selected into DC
99  memset(&lf, 0, sizeof(LOGFONT));
100  lf.lfHeight = -48;
101  lf.lfWeight = FW_BOLD;
102  lf.lfCharSet = ANSI_CHARSET;
103  lf.lfOutPrecision = OUT_DEFAULT_PRECIS;
104  lf.lfClipPrecision = CLIP_DEFAULT_PRECIS;
105  lf.lfQuality = DEFAULT_QUALITY;
106  lf.lfPitchAndFamily = FF_DONTCARE | DEFAULT_PITCH;
107  lstrcpy(lf.lfFaceName, "Arial");
108 
109  hFont = CreateFontIndirect(&lf);
110  hOldFont = SelectObject(hDC, hFont);
111 
112 // Create a set of display lists based on the glyphs of the TT font we selected
113 
114  /*
115  The wglUseFontOutlines function approximates glyph outlines
116  by subdividing the quadratic B-spline curves of the outline into line segments,
117  until the distance between the outline and the interpolated midpoint is within
118  the value specified by deviation.
119  This is the final format used when format is WGL_FONT_LINES.
120  When you specify WGL_FONT_OUTLINES, the display lists created don't contain
121  any normals; thus lighting doesn't work properly.
122  To get the correct lighting of lines use WGL_FONT_POLYGONS and set
123  glPolygonMode(GL_FRONT, GL_LINE).
124  When you specify format as WGL_FONT_POLYGONS the outlines are further
125  tessellated into separate triangles, triangle fans, triangle strips, or
126  quadrilateral strips to create the surface of each glyph.
127  With WGL_FONT_POLYGONS, the created display lists call glFrontFace(GL_CW)
128  or glFrontFace(GL_CCW); thus the current front-face value might be altered.
129  For the best appearance of text with WGL_FONT_POLYGONS, cull the back faces
130  as follows:
131 
132 
133  */
134 
135  glCullFace(GL_BACK);
136  glEnable(GL_CULL_FACE);
137 
138  // Create a set of display lists based on the glyphs of the TT font we selected
139  if (!(wglUseFontOutlines(hDC, 0, 127, GLF_START_LIST, 0.0f,
140  m_glOptions.textExtrusion, WGL_FONT_POLYGONS, glyphMetrics)))
141  {
142  AnsiString error = AnsiString("wglUseFontOutlines failed!")
143  + SysErrorMessage(GetLastError());
144  OutputDebugString(error.c_str());
145  throw error.c_str();
146  }
147  DeleteObject(SelectObject(hDC, hOldFont));
148

272 void AboutGLDisplay::drawTorus(void)
273 {
274  struct vertex
275  {
276  GLfloat t[2];
277  GLfloat n[3];
278  GLfloat v[3];
279  };
280 
281  int numVerts = (objectNumMajor + 1) * (objectNumMinor + 1);
282  int numStrips =
283  m_glOptions.halfObject ? objectNumMajor / 2 : objectNumMajor;
284  int numPerStrip = 2 * (objectNumMinor + 1);
285  int numElements = (objectNumMajor + 1) * numPerStrip;
286 
287  static struct vertex *vertexArray, *v;
288  static GLuint *elementArray, *e;
289  static int numMajor;
290  static int numMinor;
291  int i, j;
292 
293  if (!vertexArray || numMajor != objectNumMajor
294  || numMinor != objectNumMinor)
295  {
296  float majorRadius = 0.6F;
297  float minorRadius = 0.2F;
298  double majorStep = 2.0F * M_PI / objectNumMajor;
299  double minorStep = 2.0F * M_PI / objectNumMinor;
300 
301  if (vertexArray)
302  free(vertexArray);
303  vertexArray = (struct vertex *) calloc(numVerts, sizeof(struct vertex));
304 
305  if (elementArray)
306  free(elementArray);
307  elementArray = (GLuint *) calloc(numElements, sizeof(GLuint));
308 
309  numMajor = objectNumMajor;
310  numMinor = objectNumMinor;
311 
312  v = vertexArray;
313  e = elementArray;
314  for (i = 0; i <= numMajor; ++i)
315  {
316  double a = i * majorStep;
317  GLfloat x = (GLfloat) cos(a);
318  GLfloat y = (GLfloat) sin(a);
319 
320  for (j = 0; j <= numMinor; ++j)
321  {
322  double b = j * minorStep;
323  GLfloat c = (GLfloat) cos(b);
324  GLfloat r = minorRadius * c + majorRadius;
325  GLfloat z = minorRadius * (GLfloat) sin(b);
326 
327  v->t[0] = i / (GLfloat) numMajor;
328  v->t[1] = j / (GLfloat) numMinor;
329 
330  v->n[0] = x * c;
331  v->n[1] = y * c;
332  v->n[2] = z / minorRadius;
333 
334  v->v[0] = x * r;
335  v->v[1] = y * r;
336  v->v[2] = z;
337 
338  v++;
339 
340  *e++ = i * (numMinor + 1) + j;
341  *e++ = (i + 1) * (numMinor + 1) + j;
342  }
343  }
344  }
345 
346  if (m_glOptions.useVertexArray)
347  {
348  glInterleavedArrays(GL_T2F_N3F_V3F, 0, vertexArray);
349 
350 #if defined(GL_SGI_compiled_vertex_array)
351  if (useVertexLocking && LockArrays) LockArrays(0, numVerts);
352 #endif
353 
354  for (i = 0, e = elementArray; i < numStrips; ++i, e += numPerStrip)
355  {
356  glDrawElements(GL_TRIANGLE_STRIP, numPerStrip, GL_UNSIGNED_INT, e);
357  }
358 
359 #if defined(GL_SGI_compiled_vertex_array)
360  if (useVertexLocking && UnlockArrays) UnlockArrays();
361 #endif
362  }
363  else
364  {
365  for (i = 0, e = elementArray; i < numStrips; ++i, e += numPerStrip)
366  {
367  glBegin(GL_TRIANGLE_STRIP);
368  for (j = 0; j < numPerStrip; ++j)
369  {
370  v = &vertexArray[e[j]];
371 
372  glTexCoord2fv(v->t);
373  glNormal3fv(v->n);
374  glVertex3fv(v->v);
375  }
376  glEnd();
377  }
378  }

779 void AboutGLDisplay::ptov(int x, int y, int width, int height, float v[3])
780 {
781  float d, a;
782 
783  // project x,y onto a hemi-sphere centered within width, height
784  v[0] = (2.0F * x - width) / width;
785  v[1] = (height - 2.0F * y) / height;
786  d = (float) sqrt(v[0] * v[0] + v[1] * v[1]);
787  v[2] = (float) cos((M_PI / 2.0F) * ((d < 1.0F) ? d : 1.0F));
788  a = 1.0F / (float) sqrt(v[0] * v[0] + v[1] * v[1] + v[2] * v[2]);
789  v[0] *= a;
790  v[1] *= a;
791  v[2] *= a;
792 }

854 void AboutGLDisplay::setupPalette()
855 {
856  PIXELFORMATDESCRIPTOR pfd;
857  LOGPALETTE* pPal;
858  int pixelFormat = GetPixelFormat(hDC);
859  int paletteSize;
860 
861  DescribePixelFormat(hDC, pixelFormat, sizeof(PIXELFORMATDESCRIPTOR), &pfd);
862  if (!((pfd.dwFlags & PFD_NEED_PALETTE)
863  || (pfd.iPixelType == PFD_TYPE_COLORINDEX)))
864  {
865  return;
866  }
867 
868  paletteSize = 1 << pfd.cColorBits;
869  pPal = (LOGPALETTE*) malloc(
870  sizeof(LOGPALETTE) + paletteSize * sizeof(PALETTEENTRY));
871  pPal->palVersion = 0x300;
872  pPal->palNumEntries = (WORD) paletteSize;
873 
874  // start with a copy of the current system palette
875  (void) GetSystemPaletteEntries(hDC, 0, paletteSize, &pPal->palPalEntry[0]);
876 
877  {
878  // fill in an RGBA color palette
879  int redMask = (1 << pfd.cRedBits) - 1;
880  int greenMask = (1 << pfd.cGreenBits) - 1;
881  int blueMask = (1 << pfd.cBlueBits) - 1;
882  int i;
883 
884  for (i = 0; i < paletteSize; ++i)
885  {
886  pPal->palPalEntry[i].peRed = (GLbyte) ((((i >> pfd.cRedShift)
887  & redMask) * 255) / redMask);
888  pPal->palPalEntry[i].peGreen = (GLbyte) ((((i >> pfd.cGreenShift)
889  & greenMask) * 255) / greenMask);
890  pPal->palPalEntry[i].peBlue = (GLbyte) ((((i >> pfd.cBlueShift)
891  & blueMask) * 255) / blueMask);
892  pPal->palPalEntry[i].peFlags = 0;
893  }
894  }
895 
896  HPALETTE hPalette;
897  hPalette = CreatePalette(pPal);
898  free(pPal);
899 
900  if (hPalette)
901  {
902  SelectPalette(hDC, hPalette, FALSE);
903  RealizePalette(hDC);
904  }
905 }
