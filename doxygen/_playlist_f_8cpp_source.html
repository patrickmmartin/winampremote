234 void __fastcall TfrmPlaylist::lstSongsDragDrop(TObject *, TObject *, int X, int Y)
235 {
236 int i;
237 
238 char filename[RET_STR_SIZE];
239 char saveident[RET_STR_SIZE];
240 bool CurrentSong;
241 int NewPos;
242 
243  // want to know if exceed the list length
244  int DropIndex = lstSongs->ItemAtPos(TPoint(X, Y), false);
245 
246 
247  TStringList * TopList = new TStringList;
248  TStringList * MiddleList = new TStringList;
249  TStringList * BottomList = new TStringList;
250  try
251  {
252  /* rather wasteful, as we have to get all the undeleted items and resend them to winamp*/
253 
254  // cache this
255  strcpy(saveident, frmMain->IdentChars);
256 
257  for (i = 0 ; i < lstSongs->Items->Count; i++)
258  {
259  strcpy(filename, saveident);
260  StringResult(filename, IPC_GETPLAYLISTFILE, i);
261  CurrentSong = i == _currentpos;
262 
263  if (lstSongs->Selected[i])
264  {
265  MiddleList->AddObject(filename, (TObject *) CurrentSong);
266  }
267  else if (i < DropIndex)
268  {
269  TopList->AddObject(filename, (TObject *) CurrentSong);
270  }
271  else
272  {
273  BottomList->AddObject(filename, (TObject *) CurrentSong);
274  }
275  } // for
276 
277  // delete
278  frmMain->Delete->Execute();
279 
280  // rebuild list
281  TopList->AddStrings(MiddleList);
282  TopList->AddStrings(BottomList);
283  // reset position
284 
285  frmMain->DoAddFiles(TopList);
286 
287  NewPos = TopList->IndexOfObject((TObject *) true);
288  IntegerResult(frmMain->IdentChars, IPC_SETPLAYLISTPOS, NewPos);
289 
290  frmMain->PlaylistRefresh->Execute();
291  }
292  __finally
293  {
294  delete BottomList;
295  delete MiddleList;
296  delete TopList;
297  }
298 }
